import numpy as np import matplotlib.pyplot as plt import matplotlib.animation as animation from matplotlib.widgets import Slider, Button, RadioButtons import time # ========================================== # 1. 물리 상수 및 위성 데이터 # ========================================== MU = 3.986e14 # 지구 중력 상수 (m^3/s^2) RE = 6378000 # 지구 반지름 (m) H0 = 400000 # 초기 고도 (m) R0 = RE + H0 # 초기 궤도 반경 (m) V0 = np.sqrt(MU / R0) # 초기 원궤도 선속도 (m/s) RHO0 = 1.225 # 해수면 대기 밀도 (kg/m^3) H_SCALE = 7500 # 대기 스케일 하이트 (m) R_NEMO = 2688000 # 포인트 니모 합격 반경 (m) ATM_H = 100000 # 대기권 진입 기준 고도 (m) SATELLITES = { 'Sputnik 1': {'m': 83.6, 'A': 0.26, 'Cd': 2.1, 'tau': 1.0, 'k': 5000}, 'Mir': {'m': 130000, 'A': 350, 'Cd': 2.3, 'tau': 2.0, 'k': 150000}, 'ISS': {'m': 450000, 'A': 2200, 'Cd': 2.6, 'tau': 3.0, 'k': 400000} } # ========================================== # 2. 시뮬레이션 클래스 정의 # ========================================== class DeorbitSimulation: def __init__(self): self.fig, self.ax = plt.subplots(figsize=(8, 8)) self.fig.subplots_adjust(bottom=0.3) self.ax.set_aspect('equal') self.ax.axis('off') # UI 상태 변수 self.state = 'IDLE' self.sat_name = 'Sputnik 1' self.dv = 10.0 # 물리 상태 변수 self.x, self.y, self.vx, self.vy = 0, 0, 0, 0 self.orbit_start_time = 0 self.deorbit_start_time = 0 self.entered_atm = False self.L_final = 0 self.setup_visuals() self.setup_ui() self.anim = animation.FuncAnimation(self.fig, self.update, interval=30, blit=False, cache_frame_data=False) def setup_visuals(self): self.ax.set_xlim(-(R0 + 500000), R0 + 500000) self.ax.set_ylim(-(R0 + 500000), R0 + 500000) # 지구 (파란색) earth = plt.Circle((0, 0), RE, color='#1f77b4', zorder=1) self.ax.add_patch(earth) # 대기권 (하늘색, 투명도 적용) atm = plt.Circle((0, 0), RE + ATM_H, color='lightblue', alpha=0.2, zorder=0) self.ax.add_patch(atm) # 포인트 니모 구역 (초록색 호) # 타겟 센터를 오른쪽(theta=0)으로 설정 nemo_angles = np.linspace(-R_NEMO/RE, R_NEMO/RE, 100) self.ax.plot(RE * np.cos(nemo_angles), RE * np.sin(nemo_angles), color='lime', linewidth=5, zorder=2) # 위성/파편 그래픽 객체 self.sat_plot, = self.ax.plot([], [], 'o', color='black', markersize=6, zorder=3) # 결과 텍스트 (성공/실패) self.result_text = self.ax.text(0, 0, "", fontsize=30, fontweight='bold', color='red', ha='center', va='center', zorder=4) def setup_ui(self): # 라디오 버튼 (위성 선택) ax_radio = plt.axes([0.1, 0.1, 0.25, 0.15]) self.radio = RadioButtons(ax_radio, ('Sputnik 1', 'Mir', 'ISS')) self.radio.on_clicked(self.set_satellite) # 슬라이더 (감속량) ax_slider = plt.axes([0.45, 0.2, 0.4, 0.03]) self.slider = Slider(ax_slider, '$\Delta v$', 10.0, 300.0, valinit=10.0) self.slider.on_changed(self.set_dv) # Start 버튼 ax_start = plt.axes([0.45, 0.1, 0.15, 0.05]) self.btn_start = Button(ax_start, 'Start') self.btn_start.on_clicked(self.start_sim) # Reset 버튼 ax_reset = plt.axes([0.65, 0.1, 0.15, 0.05]) self.btn_reset = Button(ax_reset, 'Reset') self.btn_reset.on_clicked(self.reset_sim) def set_satellite(self, label): self.sat_name = label def set_dv(self, val): self.dv = val def start_sim(self, event): if self.state in ['IDLE', 'DONE']: self.state = 'ORBITING' self.orbit_start_time = time.time() self.result_text.set_text("") self.entered_atm = False self.sat_plot.set_marker('o') self.sat_plot.set_linestyle('') self.sat_plot.set_color('black') def reset_sim(self, event): self.state = 'IDLE' self.result_text.set_text("") self.sat_plot.set_data([], []) def rk4_step(self, dt): sat = SATELLITES[self.sat_name] def derivatives(state): sx, sy, svx, svy = state r = np.hypot(sx, sy) v = np.hypot(svx, svy) # 중력 가속도 ax = -MU * sx / r**3 ay = -MU * sy / r**3 # 대기 저항 가속도 if r < RE + ATM_H: h = r - RE rho = RHO0 * np.exp(-h / H_SCALE) drag = 0.5 * rho * v * (sat['Cd'] * sat['A'] / sat['m']) ax -= drag * svx / v ay -= drag * svy / v return np.array([svx, svy, ax, ay]) s0 = np.array([self.x, self.y, self.vx, self.vy]) k1 = derivatives(s0) k2 = derivatives(s0 + k1 * dt / 2) k3 = derivatives(s0 + k2 * dt / 2) k4 = derivatives(s0 + k3 * dt) new_state = s0 + (dt / 6) * (k1 + 2*k2 + 2*k3 + k4) self.x, self.y, self.vx, self.vy = new_state def update(self, frame): if self.state == 'IDLE' or self.state == 'DONE': return self.sat_plot, self.result_text # 1. 궤도 진입 (포인트 니모 대척점으로 이동) if self.state == 'ORBITING': elapsed = time.time() - self.orbit_start_time tau = SATELLITES[self.sat_name]['tau'] # 위에서 출발하여 tau초 만에 반대편(대척점, theta=pi)으로 이동 progress = min(elapsed / tau, 1.0) theta = np.pi/2 + progress * (np.pi/2) # 90도에서 180도(대척점)로 이동 self.x = R0 * np.cos(theta) self.y = R0 * np.sin(theta) self.sat_plot.set_data([self.x], [self.y]) if elapsed >= tau: self.state = 'DEORBITING' self.deorbit_start_time = time.time() # 대척점(왼쪽)에서 역추진 적용 (반시계 방향 낙하) v_new = V0 - self.dv self.x, self.y = -R0, 0 self.vx, self.vy = 0, -v_new # 2. 역추진 낙하 시뮬레이션 (물리 엔진 적용) elif self.state == 'DEORBITING': # 2초 타임아웃 룰 적용 (무한 궤도 비행 방지) if time.time() - self.deorbit_start_time > 2.0: self.result_text.set_text("FAIL") self.result_text.set_color("red") self.state = 'DONE' return self.sat_plot, self.result_text # 빠른 시각화를 위한 서브스텝 처리 dt = 2.0 substeps = 50 for _ in range(substeps): self.rk4_step(dt) r = np.hypot(self.x, self.y) # 3. 대기권 진입 순간 처리 (진입각 및 최종 호 길이 확정) if r < RE + ATM_H and not self.entered_atm: self.entered_atm = True v_sq = self.vx**2 + self.vy**2 v_mag = np.sqrt(v_sq) E = v_sq / 2 - MU / r v_entry = np.sqrt(max(0, 2 * (E + MU / (RE + ATM_H)))) h_ang = self.x * self.vy - self.y * self.vx v_h = h_ang / (RE + ATM_H) ratio = max(-1.0, min(1.0, abs(v_h) / v_entry)) gamma_entry = max(np.arccos(ratio), 1e-6) self.L_final = SATELLITES[self.sat_name]['k'] / np.sin(gamma_entry) # 시각적 변화 트리거 self.sat_plot.set_marker('') self.sat_plot.set_linestyle('-') self.sat_plot.set_color('orange') self.sat_plot.set_linewidth(3) # 4. 바다(지표면) 도달 및 판정 if r <= RE: theta_center = np.arctan2(self.y, self.x) theta_center = (theta_center + np.pi) % (2*np.pi) - np.pi dtheta = self.L_final / RE arc_min = theta_center - dtheta/2 arc_max = theta_center + dtheta/2 nemo_max = R_NEMO / RE nemo_min = -nemo_max if arc_min >= nemo_min and arc_max <= nemo_max: self.result_text.set_text("SUCCESS") self.result_text.set_color("lime") else: self.result_text.set_text("FAIL") self.result_text.set_color("red") self.state = 'DONE' break #
