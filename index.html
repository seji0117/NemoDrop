import math import time from pyodide.ffi import create_proxy from js import document, window # 1. 물리 정수 mu = 3.986e14 Re = 6378000 h0 = 400000 r0 = Re + h0 v0 = math.sqrt(mu / r0) Rnemo = 2688000 theta_nemo = math.pi / 2 theta_antipode = 3 * math.pi / 2 nemo_half_angle = Rnemo / Re # 2. 위성 제원 satellites = { "sputnik": {"m0": 83.6, "A": 0.26, "Cd": 2.1, "life": 1.0, "k_break": 5000}, "mir": {"m0": 130000, "A": 350, "Cd": 2.3, "life": 2.0, "k_break": 150000}, "iss": {"m0": 450000, "A": 2200, "Cd": 2.6, "life": 3.0, "k_break": 400000} } # 3. 상태 변수 sim_state = "IDLE" sim_time = 0 start_burn_real_time = 0 px, py, vx, vy = 0, 0, 0, 0 entry_gamma = 0 Lfinal = 0 arc_span = 0 first_atm = True # 4. 캔버스 렌더링 컨텍스트 canvas = document.getElementById("simCanvas") ctx = canvas.getContext("2d") cx, cy = 300, 300 scale = 180 / r0 # 지구 반지름에 맞춰 스케일링 def draw(): ctx.clearRect(0, 0, canvas.width, canvas.height) # 지구 ctx.beginPath() ctx.arc(cx, cy, Re * scale, 0, 2 * math.pi) ctx.fillStyle = "#1e293b" ctx.fill() ctx.strokeStyle = "#334155" ctx.lineWidth = 2 ctx.stroke() # 대기권 ctx.beginPath() ctx.arc(cx, cy, (Re + 100000) * scale, 0, 2 * math.pi) ctx.fillStyle = "rgba(56, 189, 248, 0.05)" ctx.fill() # 포인트 니모 (녹색 호) ctx.beginPath() ctx.arc(cx, cy, Re * scale, theta_nemo - nemo_half_angle, theta_nemo + nemo_half_angle) ctx.strokeStyle = "#10b981" ctx.lineWidth = 6 ctx.stroke() # 위성 궤적 및 파편 호 if sim_state in ["ORBITING", "DEORBIT", "ATMOSPHERE"]: r = math.sqrt(px*px + py*py) if sim_state == "ATMOSPHERE": theta = math.atan2(py, px) ctx.beginPath() ctx.arc(cx, cy, r * scale, theta - arc_span/2, theta + arc_span/2) ctx.strokeStyle = "#ef4444" ctx.lineWidth = 4 ctx.stroke() else: ctx.beginPath() ctx.arc(cx + px * scale, cy + py * scale, 3, 0, 2 * math.pi) ctx.fillStyle = "#fff" ctx.fill() def update(event=None): global sim_state, sim_time, px, py, vx, vy, first_atm global entry_gamma, Lfinal, arc_span, start_burn_real_time if sim_state == "IDLE" or sim_state == "FINISHED": draw() if sim_state != "FINISHED": window.requestAnimationFrame(update_proxy) return sat_key = document.getElementById("satSelect").value sat = satellites[sat_key] dv = float(document.getElementById("dvSlider").value) substeps = 80 dt = 0.2 for _ in range(substeps): sim_time += dt if sim_state == "ORBITING": # 생명주기에 도달하면 정확히 포인트 니모의 대척점에 위치 omega = math.sqrt(mu / (r0**3)) current_theta = theta_antipode - omega * (sat["life"] - sim_time) px = r0 * math.cos(current_theta) py = r0 * math.sin(current_theta) vx = -v0 * math.sin(current_theta) vy = v0 * math.cos(current_theta) if sim_time >= sat["life"]: sim_state = "DEORBIT" start_burn_real_time = time.time() # 역추진 시작 실제 시간 기록 vMag = math.sqrt(vx*vx + vy*vy) vx = (vx / vMag) * (vMag - dv) vy = (vy / vMag) * (vMag - dv) elif sim_state in ["DEORBIT", "ATMOSPHERE"]: r = math.sqrt(px*px + py*py) h = r - Re # 2초 타임아웃 체크 (현실 컴퓨터 시간 기준) if sim_state == "DEORBIT": if time.time() - start_burn_real_time > 2.0: sim_state = "FINISHED" set_result("FAIL") break # 지표면(바다) 충돌 if h <= 0: sim_state = "FINISHED" evaluate_landing() break ga = mu / (r*r) ax = -ga * (px/r) ay = -ga * (py/r) # 대기권(100km 이하) 진입 if h <= 100000: sim_state = "ATMOSPHERE" if first_atm: vMag = math.sqrt(vx*vx + vy*vy) vr = (px*vx + py*vy) / r sin_gamma = abs(vr) / vMag if sin_gamma > 1: sin_gamma = 1 if sin_gamma < 0.0001: sin_gamma = 0.0001 entry_gamma = math.asin(sin_gamma) Lfinal = sat["k_break"] / sin_gamma arc_span = Lfinal / Re first_atm = False # 공기 저항 가속도 rho = 1.225 * math.exp(-h / 7500) vMag = math.sqrt(vx*vx + vy*vy) ad = 0.5 * rho * vMag * vMag * (sat["Cd"] * sat["A"]) / sat["m0"] ax -= ad * (vx / vMag) ay -= ad * (vy / vMag) vx += ax * dt vy += ay * dt px += vx * dt py += vy * dt draw() if sim_state != "FINISHED": window.requestAnimationFrame(update_proxy) def evaluate_landing(): landing_theta = math.atan2(py, px) if landing_theta < 0: landing_theta += 2 * math.pi angular_diff = abs(landing_theta - theta_nemo) if angular_diff > math.pi: angular_diff = 2 * math.pi - angular_diff # 파편 면적 전체가 포인트 니모 구역 안에 있는지 확인 if angular_diff + (arc_span / 2) <= nemo_half_angle: set_result("SUCCESS") else: set_result("FAIL") def set_result(text): res_div = document.getElementById("resultText") res_div.innerText = text res_div.className = text def start_sim(event): global sim_state, sim_time, first_atm if sim_state != "IDLE": return sim_state = "ORBITING" sim_time = 0 first_atm = True document.getElementById("resultText").innerText = "" window.requestAnimationFrame(update_proxy) def reset_sim(event): global sim_state, px, py, vx, vy sim_state = "IDLE" px, py, vx, vy = 0, 0, 0, 0 document.getElementById("resultText").innerText = "" draw() # 이벤트 리스너용 프록시 생성 update_proxy = create_proxy(update) start_proxy = create_proxy(start_sim) reset_proxy = create_proxy(reset_sim) document.getElementById("startBtn").addEventListener("click", start_proxy) document.getElementById("resetBtn").addEventListener("click", reset_proxy) # 초기 렌더링 draw()
